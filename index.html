async function init() { // 這裡直接使用全域變數 PIXI const app = new PIXI.Application({ view: document.getElementById("canvas"), autoStart: true, backgroundAlpha: 0, resizeTo: window }); // 透過全域插件 PIXI.live2d 載入模型 // 確保路徑與你的 GitHub 結構一致 const model = await PIXI.live2d.Live2DModel.from("public/model/model.model3.json"); app.stage.addChild(model); // 初始位置調整 model.x = 0; model.y = 0; } init();
